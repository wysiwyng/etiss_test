47 template <unsigned index>
48 static bool l_cust(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
49 {
50  switch (index)
51  {
52  case 1: // l.cust1
53  cs.append(CodePart::APPENDEDRETURNINGREQUIRED).code() = "return ETISS_RETURNCODE_CPUFINISHED ;";
54  break;
55  case 2: // l.cust2 // get cpu time
56  cs.append(CodePart::INITIALREQUIRED).code() =
57  std::string(reg[rd.read(ba)]) + " = (etiss_uint32) (cpu->cpuTime_ps/1000000);";
58  break;
59  case 3: // l.cust3
60  cs.append(CodePart::INITIALREQUIRED).code() = std::string("OR1K_printchar((char)") + reg[ra.read(ba)] + " );";
61  break;
62  case 4: // l.cust4
63  cs.append(CodePart::INITIALREQUIRED).code() = "OR1K_printtime(cpu);";
64  break;
65  case 5: // l.cust5
66  cs.append(CodePart::INITIALREQUIRED).code() = std::string(reg[rd.read(ba)]) + " = OR1K_reset_stopwatch(cpu),";
67  break;
68  default:
69  // nop
70  break;
71  }
72  update_pc<false, false>(ba, cs, ic);
73  return true;
74 }

76 static void OR1K_l_mulX(const std::string rD, const std::string rA, const std::string rB, CodeSet &codeset)

91  "((((OR1K_cy==0)&&(OR1K_high==0))||((OR1K_cy==-1)&&(OR1K_high==-1)))?0:OR1K_SR_OV);";
92  }
93  // exception
94  {
95  CodePart &part = codeset.append(CodePart::APPENDEDRETURNINGREQUIRED);
96  part.getAffectedRegisters().add("SR", 32, OR1K_SR_OV);
97  std::stringstream ss;
98  ss << "\t\tif (unlikely((*((OR1K*)cpu)->SR)&OR1K_SR_OVE)){\n";
99 
100  ss << "\t\t\tif (unlikely((*((OR1K*)cpu)->AECR) & OR1K_AECR_OVMULE)){\n";

102  ss << "etiss_int32 OR1K_high_ = OR1K_result >> 32; etiss_int32 OR1K_cy_ = OR1K_result >> "
103  "31;\n*((OR1K*)cpu)->SR = ((*((OR1K*)cpu)->SR) & ~OR1K_SR_OV) | "
104  "((((OR1K_cy_==0)&&(OR1K_high_==0))||((OR1K_cy_==-1)&&(OR1K_high_==-1)))?0:OR1K_SR_OV);";
105  ss << "\t\t\t\tif ((*((OR1K*)cpu)->SR) & OR1K_SR_OV) return OR1K_RANGEEXCEPTION;\n";
106 
107  ss << "\t\t\t}\n";
108 
109  ss << "\t\t}";
110  part.code() = ss.str();
111  }

147 template <typename T>
148 static bool l_ext(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
149 {
150  std::string cast;
151  const bool sig = std::is_signed<T>::value;
152  cast = std::string("etiss_") + (sig ? "" : "u") + "int" + toString(sizeof(T) * 8);
153 
154  CodePart &cp = cs.append(CodePart::INITIALREQUIRED);
155  cp = std::string(reg[rd.read(ba)]) + " = (" + cast + ")" + reg[ra.read(ba)] + ";";
156  cp.getRegisterDependencies().add(reg_name[ra.read(ba)], 32);
157  cp.getAffectedRegisters().add(reg_name[rd.read(ba)], 32);
158  update_pc<false, false>(ba, cs, ic);
159  return true;
160 }
