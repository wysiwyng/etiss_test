48 void OR1KArch_apply_fpu_flags(OR1K *cpu, unsigned flags)
49 {
50  etiss_uint32 val = *(cpu->FPCSR) & 0xFFFFF007;
51  if (flags & float_flag_invalid)
52  {
53  val |= OR1K_FPCSR_IVF;
54  }
55  if (flags & float_flag_divbyzero)
56  {
57  val |= OR1K_FPCSR_DZF;
58  }
59  if (val & float_flag_overflow)
60  {
61  val |= OR1K_FPCSR_OVF;
62  }
63  if (val & float_flag_underflow)
64  {
65  val |= OR1K_FPCSR_UNF;
66  }
67  if (val & float_flag_inexact)
68  {
69  val |= OR1K_FPCSR_IXF;
70  }
71  *(cpu->FPCSR) = val;
72 }
