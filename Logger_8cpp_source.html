190 int32_t Logger::log(bool isread, uint64_t local_addr, uint8_t *buf, unsigned len)
191 {
192  if (isread)
193  {
194  // memset(buf,0,len); //Produces segfault when iread is called!
195  return 0;
196  }
197 
198  if (len <= 0)
199  return 0;
200 
201  if (buf == 0)
202  return 0;
203 
204  if (len == 1)
205  std::cout << (char)buf[0];
206  else
207  {
208  // print as different types
209  int iout = 0;
210  int fbuf = 0;
211  std::cout << " <char> ";
212  for (unsigned i = 0; i < len; i++)
213  {
214  iout |= buf[i] << (len - i - 1) * 8;
215  fbuf |= buf[i] << (len - i - 1) * 8;
216  bool isChar = (buf[i] > 32 && buf[i] < 127);
217  if (buf[i] == '\n')
218  std::cout << "\\n";
219  else
220  std::cout << (isChar ? std::string("\033[1;31m") : std::string("")) << std::setw(2) << (char)buf[i]
221  << (isChar ? std::string("\033[0m") : std::string(""));
222  }
223  float *fout = reinterpret_cast<float *>(&fbuf);
224 
225  std::cout << std::setfill('0') << std::hex << " | <hex> 0x" << std::setw(8) << iout << std::dec
226  << std::setfill(' ');
227  std::cout << " | <int> " << std::setw(10) << iout;
228  std::cout << " | <float> " << *fout;
229  std::cout << std::endl;
230  }
231  /*for(unsigned i = 0; i < len; i++)
232  {
233  std::cout << (char) buf[i];
234  }*/
235 
236  std::flush(std::cout);
237 
238  return 0;
239 }
