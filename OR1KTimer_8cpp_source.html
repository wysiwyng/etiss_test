49 etiss::int32 OR1KTimer::execute()
50 {
51  // when this function is called, TTCR is not up to date.
52  // Therefore we update it according to the internal state registers
53  // last_time_0_ps_ and next_timer_match_ps_.
54  bool need_internal_update = internal_state_to_TTCR();
55 
56  // We also use the internal registers to determine whether an interrupt
57  // must be sent, as this is much easier than using TTCR and TTMPR (we just
58  // use plugin_cpu_->cpuTime_ps >= next_timer_match_ps_ as the criterion;
59  // with TTCR the problem is that we might jump over the match value if it's
60  // not at the end of a basic block)
61  bool interrupt = internal_state_to_TTMR_IP();
62 
63  // In case of a timer match, the internal variables last_time_0_ps_ and
64  // next_timer_match_ps_ are no longer up-to-date. As TTCR and TTMR are now
65  // set and up-to-date, we use those to calculate the new internal values.
66  // The "if" is just a speed optimization, it would always be possible to call
67  // regs_to_internal_state if the registers are up-to-date with the cpu time.
68  if (need_internal_update)
69  regs_to_internal_state();
70 
71  // The implementation is level-triggered: As long as TTMR_IP (interrupt
72  // pending) is set, we return OR1K_TICKTIMEREXCEPTION, which can be
73  // evaluated by ETISS if interrupts are enabled.
74  // To implement edge-triggered behaviour, save the old value of the
75  // interrupt-pending flag at the beginning of this function and
76  // return OR1K_TICKTIMEREXCEPTION only if TTMR_IP was not set before and is
77  // set now.
78  if (interrupt)
79  return OR1K_TICKTIMEREXCEPTION;
80  else
81  return etiss::RETURNCODE::NOERROR;
82 }
