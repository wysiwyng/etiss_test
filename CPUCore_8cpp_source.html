447 static void etiss_CPUCore_handleException(ETISS_CPU *cpu, etiss::int32 &code, BlockLink *&block_ptr,
448  Translation &translator, CPUArch *arch)
449 {
450 
451 #if DEBUG
452  if (unlikely(code == RETURNCODE::NOERROR))
453  {
454  etiss::log(etiss::ERROR, "etiss_CPUCore_handleException may not be called with error code NOERROR");
455  }
456 #endif
457 
458  switch (code)
459  {
460  case RETURNCODE::RELOADBLOCKS:
461  block_ptr = 0; // doesn't hold a reference and thus might become invalid
462  translator.unloadBlocks();
463  code = RETURNCODE::NOERROR;
464  return;
465  case RETURNCODE::RELOADCURRENTBLOCK:
466  if (block_ptr)
467  block_ptr->valid = false; // invalidate but don't delete block
468  block_ptr = 0;
469  code = RETURNCODE::NOERROR;
470  return;
471  case RETURNCODE::GDBNOERROR:
472  code = RETURNCODE::NOERROR;
473  return;
474  case RETURNCODE::CPUFINISHED:
475  return;
476  default:
477  code = arch->handleException(code, cpu);
478  return;
479  }

821  }

825  double simulation_time = endTime - startTime;

827  double mips = cpu_->cpuTime_ps / (float)cpu_->cpuCycleTime_ps / simulation_time / 1.0E6;

871  etiss::log(etiss::INFO, std::string("MIPS (good estimation): ") +
872  etiss::toString(instrcounter / ((double)cpu_->cpuTime_ps / 1000000.0)));
873 #endif
874 
875  // cleanup plugins
876  for (auto &p : plugins)
877  {
878  if (p)
879  {
880  p->cleanup();
881  p->plugin_cpu_ = nullptr;
882  p->plugin_system_ = nullptr;
883  p->plugin_arch_ = nullptr;

906  {
907  vcpu_->foreachField(
908  [listener](std::shared_ptr<etiss::VirtualStruct::Field> f) { f->removeListener(listener); });
909

911  }
