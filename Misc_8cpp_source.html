149  std::string tmp;
150  unsigned state = 0;
151  for (unsigned pos = 0; pos < cmdline.length(); pos++)
152  {
153  while (true)
154  {
155  switch (state)
156  {
157  case 0:
158  switch (cmdline[pos])
159  {
160  case '\'':
161  case '\"':
162  state = 1;

164  case '\t':

179  break;
