67  // internal registers of the timer
68 
69  // Instead of TTMR_TP and TTCR, we use two internal state variables:
70  // All the functionality of the Timer is in functions which translate
71  // from the external (TTMR, TTCR) to the internal (the following vars) state.
72  int64_t last_time_0_ps_ = 0; // the last time TTCR was 0
73  // instead of TTCR; can be negativ if the sw
74  // sets TTCR to a high value in the beginning
75  // It is used to calculate TTCR;
76  // next_timer_match is not enough to calculate
77  // the first 4 bits of TTCR
78 
79  int64_t next_timer_match_ps_ = 0; // instead of TTMR_TP; needs to be updated
80  // after each timer_match
81 
82  // mode: use TTMR
83  // IE: use TTMR
84  // IP: use TTMR
85 
86  // end internal registers
87 
88  // convert external state (TTCR, TTMR) to internal state (last_time_0_ps_,
89  // next_timer_match_ps_)
90  // It requires that TTCR and TTMR are up to date with the current cpu time.
91  // calculate last_time_0_ps_ and next_timer_match_ps_.
92  inline void regs_to_internal_state()
93  {
94  bool stop =
95  (getMode() == TIMER_DISABLED) || ((getMode() == TIMER_ONESHOT) && (getTTCR_28Bit() == getTTMR_TP()));
96 
97  if (!stop)
98  {
99  last_time_0_ps_ = plugin_cpu_->cpuTime_ps - getTTCR() * plugin_cpu_->cpuCycleTime_ps;
100 
101  // calculate next_timer_match_ps_
102  // Make sure TP is bigger than TTCR; I don't like negative numbers
103  uint32_t ticks_to_match = ((getTTMR_TP() + 0x10000000) - getTTCR_28Bit()) & 0x0FFFFFFF;
104  if (ticks_to_match == 0)
105  ticks_to_match = 0x10000000;
106  next_timer_match_ps_ = plugin_cpu_->cpuTime_ps + plugin_cpu_->cpuCycleTime_ps * ticks_to_match;
107  }
108  }
109 
110  // it returns true if the internal state has to be updated afterwards
111  // in order to calculate the next match time
112  // e.g. if the TTCR was changed and not only increased.
113  // This is only an optimization, regs_to_internal_state() can always be
114  // called when the registers are up-to-date with the current cpu time.
115  inline bool internal_state_to_TTCR()
116  {
117  if (getMode() != TIMER_DISABLED) // don't change TTCR if timer is disabled
118  {
119  // Do we need to change the TTCR?
120  if ((int64_t)plugin_cpu_->cpuTime_ps >= next_timer_match_ps_) // timer match
121  {
122  switch (getMode())
123  {
124  case TIMER_DISABLED: // avoid compiler warning "TIMER_DISABLED
125  return false; // not handled in switch"
126  case TIMER_AUTORESTART:
127  *((OR1K *)plugin_cpu_)->TTCR =
128  static_cast<etiss_uint32>((plugin_cpu_->cpuTime_ps - next_timer_match_ps_) / plugin_cpu_->cpuCycleTime_ps);
129  return true;
130  case TIMER_ONESHOT: // change TTCR to meet the exact value: like in hardware
131  *((OR1K *)plugin_cpu_)->TTCR = (getTTCR() & 0xf0000000) | getTTMR_TP();
132  return false;
133  case TIMER_CONTINUOUS: // just continue counting
134  *((OR1K *)plugin_cpu_)->TTCR =
135  (uint32_t)((plugin_cpu_->cpuTime_ps - last_time_0_ps_) / plugin_cpu_->cpuCycleTime_ps);
136  return true; // we need to calculate the next firing point
137  }
138  }
139  else // no timer match, just counting auto-restart, one-shot or continuous
140  {
141  *((OR1K *)plugin_cpu_)->TTCR =
142  (uint32_t)((plugin_cpu_->cpuTime_ps - last_time_0_ps_) / plugin_cpu_->cpuCycleTime_ps);
143  return false;
144  }
145  }
146  return false;
147  }
148 
149  // returns TTMR_IP
150  inline bool internal_state_to_TTMR_IP()
151  {
152  etiss::uint32 &TTMR = *((OR1K *)plugin_cpu_)->TTMR;
153 
154  bool match;
155  if (getMode() == TIMER_DISABLED)
156  match = getTTCR_28Bit() == getTTMR_TP(); // use external match in disabled case
157  else // in one-shot, we continuously fire interrupts after the match
158  match = (int64_t)plugin_cpu_->cpuTime_ps >= next_timer_match_ps_;
159 
160  // write interrupt
161  if (getTTMR_IE() && match)
162  {
163  TTMR = TTMR | 0x10000000U;
164  }
165  return TTMR & 0x10000000U;
166  }
167 
168  // simple getter helpers
169  inline Modes getMode() { return Modes(*(((OR1K *)plugin_cpu_)->TTMR) >> 30); }
170 
171  inline etiss::uint32 getTTCR() { return *((OR1K *)plugin_cpu_)->TTCR; }

180 /*
181 The timer in hardware has no additional state register, TTCR and TTMR are
182 enough.
183 
184 It has three important internal wires:
185 match
186 restart (match and auto-restart mode)
187 stop (do not increase counter: match and one-shot, disabled mode)
188 
189 In software, the implementation is much more complicated, as we don't update
190 the timer every cycle but only once in a while. Therefore, we have to remember
191 when it was last updated.
192 We could use the variable timer_last_TTCR_update_ps_ for that purpose.
193 In the older timer implementation, the variable _timer_last_ps saved the time
194 of the last time the timer was 0. Is this better? It could be problematic if
195 the software sets TTCR to a big value in the beginning, as then _timer_last_ps
196 would be negative.
197 It should not matter which of the two is used, as long as it's always updated
198 together with TTCR.
199 
200 What about saving the time of the next timer match? No risc of overflows
201 while doing comparisons for calculating timer_match!
202 Drawback: It's not possible to calculate the first 4 bits of TTCR from that.
203 For that reason, we also need the time when it last was 0.
204 But we no longer need to consider TTMR_TP, except for checking whether the timer
205 has been stopped in one-shot mode. => That's how we do it.
206 
207 The hardware model shows that it's always possible to tell from TTCR and TTMR
208 whether the timer is currently counting or not. Only if it is counting,
209 the value of the helper variables should be important; if it is not counting,
210 they should be ignored. Thus, we only have to make sure that they are always
211 updated when TTCR or TTMR are changed.
212 
213 To be compatible with the hardware model (especially while debooting OR1KVCPU,
214 that is, switching from a verilog-based RTL model back to ETISS), we should
215 be able to deduct the complete state of the timer just from TTCR and TTMR.
216 This is not completely possible; we need the debooting time as third input.
217 
218 
219 
220 
221 */
