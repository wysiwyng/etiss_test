126 template <bool isjump,bool hasdelayslot>
127 void update_pc(BitArray & ba,etiss::CodeSet & cs,InstructionContext & ic)
128 {
129 
130  {
131  CodePart & cp = cs.append(CodePart::APPENDEDOPTIONAL);
132  cp.getRegisterDependencies().add("instructionPointer",32);
133  cp.getAffectedRegisters().add("PPC",32);
134  cp.code() = "*(((OR1K*)cpu)->PPC) = (uint32_t)cpu->instructionPointer;";
135  }
136 
137 
138  if (isjump)
139  {
140  ic.force_block_end_ = true;
141  if (ic.cf_delay_slot_>0) // no jump in delay slot
142  {
143  CodePart & cp = cs.append(CodePart::APPENDEDRETURNINGREQUIRED);
144  cp.code() = "return OR1K_ILLEGALINSTRUCTIONEXCEPTION;";
145  ic.force_block_end_ = true;
146  }
147  else
148  {
149  if (hasdelayslot)
150  {
151  ic.force_append_next_instr_ = true;
152  ic.cf_delay_slot_ = 1;
153  CodePart & cp = cs.append(CodePart::APPENDEDOPTIONAL);
154  cp.getAffectedRegisters().add("instructionPointer",32);
155  cp.code() = "cpu->instructionPointer = "+toString((uint32_t)(ic.current_address_+4))+"ULL;";
156  }
157  else
158  {
159  {
160  CodePart & cp = cs.append(CodePart::APPENDEDOPTIONAL);
161  cp.getAffectedRegisters().add("instructionPointer",32);
162  cp.code() = "cpu->instructionPointer = ((OR1K*)cpu)->instructionPointer_delayed;";
163  }
164  {
165  CodePart & cp = cs.append(CodePart::APPENDEDRETURNINGREQUIRED);
166  cp.code() = "return 0;";
167  }
168  }
169  }
170  }
171  else
172  {
173  if (ic.cf_delay_slot_>0) // instruction in delay slot // only one delay slot
174  {
175  ic.cf_delay_slot_ = 0;
176  ic.force_block_end_ = true;
177  {
178  CodePart & cp = cs.append(CodePart::APPENDEDOPTIONAL);
179  cp.getAffectedRegisters().add("instructionPointer",32);
180  cp.code() = "cpu->instructionPointer = ((OR1K*)cpu)->instructionPointer_delayed;";
181  }
182  {
183  CodePart & cp = cs.append(CodePart::APPENDEDRETURNINGREQUIRED);
184  cp.code() = "return 0;";
185  }
186  }
187  else
188  {
189  CodePart & cp = cs.append(CodePart::APPENDEDOPTIONAL);
190  cp.getAffectedRegisters().add("instructionPointer",32);
191  cp.code() = "cpu->instructionPointer = "+toString((uint32_t)(ic.current_address_+4))+"ULL;";
192  }
193  }

204 template <bool sig,bool imm,char op,char eop = ' '>
205 static bool l_sf (BitArray & ba,etiss::CodeSet & cs,InstructionContext & ic)
206 {
207  std::string cast = sig?"(etiss_int32)":"(etiss_uint32)";
208  typedef typename std::conditional<sig,int16_t,uint16_t>:: type IC;
209  std::stringstream ss;
210  ss << "*((OR1K*)cpu)->SR = ((*((OR1K*)cpu)->SR) & (~OR1K_SR_F));";
211  ss << " *((OR1K*)cpu)->SR |= (" << cast << reg[ra.read(ba)] << op << eop << cast <<((!imm)?std::string(reg[rb.read(ba)]):toString((IC)immediate.read(ba))) << ")?OR1K_SR_F:0;";
212 
213  CodePart & cp =cs.append(CodePart::INITIALREQUIRED);
214  cp.getRegisterDependencies().add(reg_name[ra.read(ba)],32);
215  if (!imm)
216  cp.getRegisterDependencies().add(reg_name[rb.read(ba)],32);
217  cp.getAffectedRegisters().add("SR",32,OR1K_SR_F);
218  cp.code() = ss.str();
219 
220  update_pc<false,false>(ba,cs,ic);
221 
222  return true;
223 }
