172  ETISS_PLUGIN_EXPORT void OR1KArch_mtspr(OR1K *cpu, etiss_uint32 id, etiss_uint32 value)
173  {
174  const char *ename = 0;
175  switch ((id >> 11) & 0x1F)
176  {
177  case 0:
178  switch (id & 0x07FF)
179  {
180  case 17: // SR
181  if (!(cpu->SPR[0][17] & OR1K_SR_SM))
182  { // prevent switching from user to supervisor mode
183  value = value & ~OR1K_SR_SM;
184  }
185  value |= OR1K_SR_FO; // ensure fixed bit
186  break;
187  }
188  break;
189  case 10: // timer
190  switch (id & 0x07FF)
191  {
192  case 0:
193  ename = "TTMR";
194  // you cannot write to IP if it was 0 before.
195  // It is allowed to write if it was 1 in order to clear the
196  // interrupt.
197  // Maybe we should allow it to be able to deboot from RTL.
198  // if (!(cpu->SPR[10][0] & 0x10000000U))
199  //{
200  value = value & ~0x10000000U; // write 0 as it was before
201  //}
202  break;
203  case 1:
204  ename = "TTCR";
205  break;
206  }
207  break;
208  }
209  cpu->SPR[(id >> 11) & 0x1F][id & 0x07FF] = value;
210  if (ename)
211  CPUArch::signalChangedRegisterValue((ETISS_CPU *)cpu, ename);
212  }

374 etiss::int32 OR1KArch::handleException(etiss::int32 code, ETISS_CPU *cpu)
375 {
376 
377  // handle global known exceptions
378  switch (code)
379  {
380  case RETURNCODE::RESET:
381  cpu->instructionPointer = 0x100 + (*((OR1K *)cpu)->EVBAR >> 12);
382  return RETURNCODE::NOERROR;
383  }
384 
385  // translate exception
386  if (code < 0) // ETISS return codes are negative
387  {
388  switch (code)
389  {
390  case RETURNCODE::DBUS_READ_ERROR:
391  case RETURNCODE::DBUS_WRITE_ERROR:
392  case RETURNCODE::IBUS_READ_ERROR:
393  case RETURNCODE::IBUS_WRITE_ERROR:
394  code = OR1K_BUSEXCEPTION;
395  break;
396  case RETURNCODE::INTERRUPT:
397  code = OR1K_EXTERNALINTERRUPTEXCEPTION;
398  break;
399  case RETURNCODE::ILLEGALINSTRUCTION:
400  code = OR1K_ILLEGALINSTRUCTIONEXCEPTION;
401  etiss::log(etiss::INFO, "OR1K received illegal instruction!");
402  break;
403  case RETURNCODE::ILLEGALJUMP:
404  *((OR1K *)cpu)->PC = (*((OR1K *)cpu)->PC) & ~(uint32_t)3;
405  etiss::log(etiss::WARNING, "OR1K received illegal jump! Trying to ignore the lowest 2 PC bits!");
406  return RETURNCODE::NOERROR;
407  default:
408  return code;
409  }
410  }
411  // filter unknown
412  // uint16 arg = code&0x0000FFFF; currently not used [-Werror=unused-variable]
413 
414  // code>>16 translate OR1K exception code to jump address
415  // EVBAR is the exception vector base address register which contains an offset to the exception vector
416  uint32 vec = ((code >> 16) & 0x0000FFFF) + (*((OR1K *)cpu)->EVBAR >> 12); // TODO: should be shifted by 13?
417 
418  /* Here only the following exceptions are performed:
419  0x100 Reset,
420  0x200 Bus Error,
421  0x300 Data Page Fault,
422  0x400 Instruction Page Fault,
423  0x500 Tick Timer,
424  0x600 Alignment,
425  0x700 Illegal Instruction,
426  0x800 External Interrupt,
427  0x900 D-TLB Miss,
428  0xA00 I-TLB Miss,
429  0xB00 Range,
430  0xC00 System Call,
431  0xD00 Floating Point,
432  0xE00 Trap
433  All others will be return to ETISS*/
434  if (((vec) >= 0x0F00) || ((vec & 0x0000FF00) != vec)) // TODO: shouldn't EVBAR be added after this?
435  {
436  std::cout << "Exception code returned: " << std::hex << code << std::dec << std::endl;
437  return code;
438  }
439 
440  if (((OR1K *)cpu)->exceptionDepth != 0)
441  {
442  // std::cout << "OR1K exception ignored: " << std::hex << vec << std::dec << " time_ps: " << (cpu->cpuTime_ps)
443  // << std::endl; return RETURNCODE::NOERROR;
444  }
445  else
446  {
447  // std::cout << "OR1K exception occured: " << std::hex << vec << std::dec << " time_ps: " << (cpu->cpuTime_ps)
448  // << std::endl;
449  }
450 
451  // check if exception should be ignored
452  switch (vec)
453  {
454  case 0x0500: // timer
455  if (!ignore_sr_iee)
456  if (!((*((OR1K *)cpu)->SR) & OR1K_SR_TEE))
457  return RETURNCODE::NOERROR;
458  break;
459  case 0x0800: // interrupt
460  // cpu->instructionPointer = vec;
461  if (!((*((OR1K *)cpu)->SR) & OR1K_SR_IEE))
462  return RETURNCODE::NOERROR;
463  break;
464  }
465 
466  if (*((OR1K *)cpu)->SR & OR1K_SR_EPH) // check for high exception vector starting at address 0xF0000000
467  vec = vec | 0xF0000000;
468 
469  ((OR1K *)cpu)->exceptionDepth++;
470 
471  // trigger exception
472 
473  // store pc/sr
474  ((OR1K *)cpu)->EPCR_16[0] = ((etiss::uint32) * ((OR1K *)cpu)->PC);
475  ((OR1K *)cpu)->ESR_16[0] = *((OR1K *)cpu)->SR;
476  // update sr
477  uint32 flags = OR1K_SR_DME | OR1K_SR_IME | OR1K_SR_IEE | OR1K_SR_TEE;
478  *((OR1K *)cpu)->SR = ((*((OR1K *)cpu)->SR) & (~flags)) | OR1K_SR_SM;
479  // update pc
480  *((OR1K *)cpu)->PC = (vec);
481 
482  return RETURNCODE::NOERROR;
483 }

510 static std::string getPrettyName(uint16_t sprid)
511 {
512 
513  static bool initialized = false;
514  static std::map<uint16_t, std::string> pname_;
515  if (!initialized)
516  {
517  initialized = true;
518 
519  pname_[(0 << 11) | 17] = "SR";
520  pname_[(10 << 11) | 0] = "TTMR";
521  pname_[(10 << 11) | 1] = "TTCR";

526  return "";
527  return find->second;

655 std::shared_ptr<etiss::VirtualStruct> OR1KArch::getVirtualStruct(ETISS_CPU *cpu)
656 {
657  auto ret = etiss::VirtualStruct::allocate(cpu, [](VirtualStruct::Field *f) {
658  // delete f;
659  });
660 
661  for (uint32_t i = 0; i < (1 << 16); ++i)
662  {
663 
664  ret->addField(new SPRField(*ret, (uint16_t)i));
665  }
666
