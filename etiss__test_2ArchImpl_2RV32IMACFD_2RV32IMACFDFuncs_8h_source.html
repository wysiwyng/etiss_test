151 static inline etiss_uint32 csr_read(ETISS_CPU * const cpu, ETISS_System * const system, void * const * const plugin_pointers, etiss_uint32 csr)
152 {
153 if (csr == 1U) {
154 return *((RV32IMACFD*)cpu)->CSR[3U] & 31U;
155 }
156 if (csr == 2U) {
157 return (*((RV32IMACFD*)cpu)->CSR[3U] >> 5U) & 7U;
158 }
159 if (csr == 3072U) {
160 return etiss_get_cycles(cpu, system, plugin_pointers);
161 }
162 if (csr == 3200U) {
163 return etiss_get_cycles(cpu, system, plugin_pointers) >> 32U;
164 }
165 if (csr == 3073U) {
166 return etiss_get_time();
167 }
168 if (csr == 3201U) {
169 return etiss_get_time() >> 32U;
170 }
171 if (csr == 3074U) {
172 return etiss_get_instret(cpu, system, plugin_pointers);
173 }
174 if (csr == 3202U) {
175 return etiss_get_instret(cpu, system, plugin_pointers) >> 32U;
176 }
177 return *((RV32IMACFD*)cpu)->CSR[csr];
178 }

183 {

190 } else {
