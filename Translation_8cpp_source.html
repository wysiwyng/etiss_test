147 void **Translation::init()
148 {
149  delete[] plugins_array_;
150  plugins_array_ = 0;
151  delete[] plugins_handle_array_;
152  plugins_handle_array_ = 0;
153  delete mis_;
154  mis_ = 0;
155 
156  if (arch_ == nullptr)
157  return nullptr;
158 
159  if (jit_ == nullptr)
160  return nullptr;
161 
162  // First plugin to more to tmpl list is the architecture plugin
163  mis_ = new etiss::instr::ModedInstructionSet(arch_->getName());
164 
165  std::vector<etiss::TranslationPlugin *> tmpl;
166 
167  tmpl.push_back(arch_);
168 
169  // Followed by all plugins, which are of type translation plugin
170  for (auto iter = plugins_.begin(); iter != plugins_.end(); iter++)
171  {
172  etiss::Plugin *p = iter->get();
173  if (p != nullptr)
174  {
175  etiss::TranslationPlugin *tp = p->getTranslationPlugin();
176  if (tp != nullptr)
177  {
178  tmpl.push_back(tp);
179  }
180  }
181  }
182 
183  // Build up the plugin lists
184  plugins_array_ = new etiss::TranslationPlugin *[tmpl.size() + 1];
185  plugins_array_[tmpl.size()] = nullptr;
186  plugins_handle_array_ = new void *[tmpl.size()];
187  plugins_array_size_ = tmpl.size();
188 
189  for (size_t i = 0; i < tmpl.size(); i++)
190  {
191  plugins_array_[i] = tmpl[i];
192  plugins_handle_array_[i] = plugins_array_[i]->getPluginHandle();
193  /* During execution of each code block, the list plugin_handle_, which
194  returned by the function init, is passed again as parameter with name
195  plugin_pointers
196  -> plugin_pointers is the name of the list of translation plugin in the
197  generated c-code of all code blocks
198  */
199  plugins_array_[i]->pointerCode = std::string("(plugin_pointers[") + toString(i) + "])";
200  }
201 
202  // Call the function initInstrSet for all translation plugins
203  for (size_t i = 0; i < tmpl.size(); i++)
204  {
205  plugins_array_[i]->initInstrSet(*mis_);
206  }
207  // Call the function finalizeInstrSet( for all translation plugins
208  for (int i = (int)tmpl.size() - 1; i >= 0; i--)
209  {
210  plugins_array_[i]->finalizeInstrSet(*mis_);
211  }
212 
214 
215  if (!mis_->compile())
216  {
217  etiss::log(etiss::ERROR, "Failed to compile instruction set");
218  delete[] plugins_array_;
219  plugins_array_ = 0;
220  delete[] plugins_handle_array_;
221  plugins_handle_array_ = 0;
222  delete mis_;
223  mis_ = 0;
224  }
225 
226  // Builds the function for function pointer plugins_initCodeBlock_ It calls_initCodeBlock functions of all
227  // translation plugins Builds the functions plugins_finalizeCodeBlock_ It calls_finalizeCodeBlock functions of all
228  // translation plugins
229  switch (plugins_array_size_)
230  {
231  case 1:
232  plugins_initCodeBlock_ = &(call_initCodeBlock<1, 0>);
233  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<1, 0>);
234  break;
235  case 2:
236  plugins_initCodeBlock_ = &(call_initCodeBlock<2, 0>);
237  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<2, 0>);
238  break;
239  case 3:
240  plugins_initCodeBlock_ = &(call_initCodeBlock<3, 0>);
241  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<3, 0>);
242  break;
243  case 4:
244  plugins_initCodeBlock_ = &(call_initCodeBlock<4, 0>);
245  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<4, 0>);
246  break;
247  case 5:
248  plugins_initCodeBlock_ = &(call_initCodeBlock<5, 0>);
249  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<5, 0>);
250  break;
251  case 6:
252  plugins_initCodeBlock_ = &(call_initCodeBlock<6, 0>);
253  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock<6, 0>);
254  break;
255  default:
256  plugins_initCodeBlock_ = &(call_initCodeBlock_ul);
257  plugins_finalizeCodeBlock_ = &(call_finalizeCodeBlock_ul);
258  }
259 
260  return plugins_handle_array_;

263 BlockLink *Translation::getBlock(BlockLink *prev, const etiss::uint64 &instructionindex)
264 {
265 
266  std::string error;
267 
268  if (prev != 0 && !prev->valid)
269  {
270  prev = 0;
271  }
272 
273  // search block in cache
274  std::list<BlockLink *> &list = blockmap_[instructionindex >> 9];
275  for (std::list<BlockLink *>::iterator iter = list.begin(); iter != list.end();) // iter++ moved into block
276  {
277  BlockLink *iterbl = (*iter);
278  if (iterbl != 0)
279  {
280  if (iterbl->valid) // check for valid block
281  {
282  if (iterbl->start <= instructionindex && iterbl->end > instructionindex)
283  {
284  if (prev != 0)
285  {
286  if (prev->end == iterbl->start)
287  {
288  BlockLink::updateRef(prev->next, iterbl);
289  }
290  else
291  {
292  BlockLink::updateRef(prev->branch, iterbl);
293  }
294  }
295  return *iter;
296  }
297  iter++;
298  }
299  else // cleanup
300  {
301  BlockLink::updateRef(iterbl->next, 0);
302  BlockLink::updateRef(iterbl->branch, 0);
303  list.erase(iter++);
304  BlockLink::decrRef(
305  iterbl); // remove reference of map // prev remains valid because this blocklink needs to be invalid
306  continue;
307  }
308  }
309  }
310 
311  // generate block
312 
313  std::string blockfunctionname;
314  {
315  std::stringstream ss;
316  ss << "_t" << id << "c" << tblockcount++ << "_block_" << instructionindex;
317  blockfunctionname = ss.str();
318  }
319 
320  CodeBlock block(instructionindex);
321  block.fileglobalCode().insert("#include \"etiss/jit/CPU.h\"\n"
322  "#include \"etiss/jit/System.h\"\n"
323  "#include \"etiss/jit/libresources.h\"\n"
324  "#include \"etiss/jit/ReturnCode.h\"\n"
325  "#include \"etiss/jit/libCSRCounters.h\"\n");
326 
327  for(auto &it: jitExtHeaders()){
328  if(it != "") block.fileglobalCode().insert("#include \"" + it + "\"\n");
329  }
330 
331  block.functionglobalCode().insert("if (cpu->mode != " + toString(cpu_.mode) +
332  ") return ETISS_RETURNCODE_RELOADCURRENTBLOCK;");
333 
334  plugins_initCodeBlock_(plugins_array_, block);
335 
336  etiss::int32 transerror = translateBlock(block);
337 
338  if (transerror != ETISS_RETURNCODE_NOERROR)
339  {
340  etiss::log(etiss::ERROR, "Failed to translate block");
341  return nullptr;
342  }
343 
344  plugins_finalizeCodeBlock_(plugins_array_, block);
345 
346  std::string code;
347  {
348  std::stringstream ss;
349  block.toCode(ss, blockfunctionname, nullptr);
350  code = ss.str();
351  }
352 
353  // various includes
354  std::set<std::string> headers;
355  headers.insert(etiss::jitFiles());
356  headers.insert(arch_->getIncludePath());
357  for(auto & it: jitExtHeaderPaths()){
358  if(it != "") headers.insert(it);
359  }
360 
361  std::set<std::string> libloc;
362  libloc.insert(arch_->getIncludePath());
363  libloc.insert(etiss::cfg().get<std::string>("etiss_path", "./"));
364  libloc.insert(etiss::jitFiles());
365  libloc.insert(etiss::jitFiles() + "/etiss/jit");
366  for(auto & it: jitExtLibPaths()){
367  if(it != "") libloc.insert(etiss::jitFiles() + it);
368  }
369 
370  std::set<std::string> libs;
371  //libs.insert("ETISS");
372  libs.insert("resources");
373  libs.insert("CSRCounters");
374  for(auto & it: jitExtLibraries()){
375  if(it != "") libs.insert(it);
376  }
377  /* DEBUG HELPER: write code files to work directory
378  {
379  static unsigned count = 0;
380  std::stringstream ss;
381  ss << "code" << ++count;
382  std::ofstream out;
383  out.open(ss.str().c_str());
384  out << code;
385  std::cout << "Code file " << count << std::endl;
386  }
387  */
388 #ifndef ETISS_DEBUG
389 #define ETISS_DEBUG 1
390 #endif
391  // compile library
392  void *funcs =
393  jit_->translate(code, headers, libloc, libs, error, etiss::cfg().get<bool>("jit.debug", ETISS_DEBUG) != 0);
394 
395  if (funcs == 0)
396  {
397  etiss::log(etiss::ERROR, error);
398  return 0;
399  }
400 
401  // wrap library handle for cleanup
402  auto local_jit = jit_;
403  std::shared_ptr<void> lib(funcs, [local_jit](void *p) { local_jit->free(p); });
404 
405  // check function/library handle
406  if (lib.get() != 0)
407  {
408  // std::cout<<"blockfunctionname:"<<blockfunctionname<<std::endl;
409  ExecBlockCall execBlock = (ExecBlockCall)jit_->getFunction(lib.get(), blockfunctionname.c_str(), error);
410  if (execBlock != 0)
411  {
412  BlockLink *nbl = new BlockLink(block.startindex_, block.endaddress_, execBlock, lib);
413  uint64 ii9 = instructionindex >> 9;
414  do
415  {
416  blockmap_[ii9].push_back(nbl);
417  BlockLink::incrRef(nbl); // map holds a reference
418  ii9++;
419  } while ((ii9 << 9) < block.endaddress_);
420 
421  if (prev != 0)
422  {
423  if (nbl->start == prev->end)
424  {
425  BlockLink::updateRef(prev->next, nbl);
426  }
427  else
428  {
429  BlockLink::updateRef(prev->branch, nbl);
430  }
431  }
432  return nbl;
433  }
434  else
435  {
436  etiss::log(etiss::ERROR, std::string("Failed to acquire function pointer from compiled library:") + error);
437  return 0;
438  }
439  }
440 
441  return 0;

445 etiss::int32 Translation::translateBlock(CodeBlock &cb)
446 {
447  cb.endaddress_ = cb.startindex_;
448 
449  etiss::instr::InstructionContext context;
450 
451  context.cf_delay_slot_ = 0;
452  context.force_block_end_ = false;
453 
454  unsigned count = 0;
455  const unsigned maxcount = etiss::cfg().get<unsigned>("etiss.max_block_size", 100);
456 
457  etiss::instr::VariableInstructionSet *const vis_ = mis_->get(cpu_.mode);
458 
459  if (vis_ == nullptr)
460  {
461  return etiss::RETURNCODE::GENERALERROR;
462  }
463 
464  etiss::instr::BitArray mainba(vis_->width_);
465 
466  do
467  {
468  context.force_append_next_instr_ = false;
469  context.force_block_end_ = false;
470  context.current_address_ = cb.endaddress_;
471  context.current_local_address_ = cb.endaddress_ - cb.startindex_;
472  context.instr_width_fully_evaluated_ = true;
473  context.is_not_default_width_ = false;
474  context.instr_width_ = vis_->width_;
475 
476  // read instruction
477  etiss::int32 ret = (*system_.dbg_read)(system_.handle, cb.endaddress_, (etiss_uint8 *)mainba.internalBuffer(),
478  mainba.byteCount()); // read instruction
479  if (ret == etiss::RETURNCODE::IBUS_READ_ERROR)
480  {
481  std::cout << "Instruction bus read error while translating!" << std::endl;
482  CodeBlock::Line &line = cb.append(cb.endaddress_); // allocate codeSet for Instruction
483  line.getCodeSet().append(CodePart::PREINITIALDEBUGRETURNING).code() =
484  std::string("cpu->cpuTime_ps += "
485  "cpu->cpuCycleTime_ps;cpu->instructionPointer += ") +
486  std::to_string(mainba.byteCount()) + std::string(";return ETISS_RETURNCODE_IBUS_READ_ERROR;");
487  // std::cout << "mainba.byteCount = " << mainba.byteCount() << std::endl;
488  cb.endaddress_ += mainba.byteCount(); // update end address
489  return etiss::RETURNCODE::NOERROR;
490  }
491  if (ret != etiss::RETURNCODE::NOERROR)
492  {
493  if (count == 0)
494  {
495  return ret; // empty block -> return error
496  }
497  else
498  {
499  break; // non empty block -> compile pending
500  }
501  }
502 
503  arch_->compensateEndianess(&cpu_, mainba);
504  // mainba.recoverFromEndianness(4,etiss::_BIG_ENDIAN_); ///TODO
505 
506  vis_->length_updater_(*vis_, context, mainba);
507 
508  // continue reading instruction data if neccessary
509  if (unlikely(context.is_not_default_width_))
510  {
511  etiss::instr::BitArray *secba = nullptr;
512  do
513  {
514  if (secba)
515  delete secba;
516  secba = new etiss::instr::BitArray(context.instr_width_);
517  ret = (*system_.dbg_read)(system_.handle, cb.endaddress_, (etiss_uint8 *)secba->internalBuffer(),
518  secba->byteCount()); // read instruction
519  if (ret != etiss::RETURNCODE::NOERROR)
520  {
521  if (count == 0)
522  {
523  delete secba;
524  return ret; // empty block -> return error
525  }
526  else
527  {
528  break; // non empty block -> compile pending
529  }
530  }
531  arch_->compensateEndianess(&cpu_, *secba);
532  // secba->recoverFromEndianness(4,etiss::_BIG_ENDIAN_);
533  vis_->length_updater_(*vis_, context, *secba);
534  } while (!context.instr_width_fully_evaluated_);
535 
536  etiss::instr::Instruction *instr;
537  etiss::instr::InstructionSet *instrSet = vis_->get(secba->width());
538  if (unlikely(!instrSet))
539  {
540  instr = &vis_->getMain()->getInvalid();
541  }
542  else
543  {
544  instr = instrSet->resolve(*secba);
545  if (unlikely(!instr))
546  {
547  instr = &instrSet->getInvalid();
548  }
549  }
550  CodeBlock::Line &line = cb.append(cb.endaddress_); // allocate codeset for instruction
551  bool ok = instr->translate(*secba, line.getCodeSet(), context);
552  if (unlikely(!ok))
553  {
554  delete secba;
555  return etiss::RETURNCODE::GENERALERROR;
556  }
557  cb.endaddress_ += secba->byteCount(); // update end address
558  delete secba;
559  }
560  else
561  {
562  etiss::instr::InstructionSet *instrSet = vis_->getMain();
563  etiss::instr::Instruction *instr = instrSet->resolve(mainba);
564  if (unlikely(instr == 0))
565  {
566  instr = &instrSet->getInvalid();
567  }
568  CodeBlock::Line &line = cb.append(cb.endaddress_); // allocate codeset for instruction
569  bool ok = instr->translate(mainba, line.getCodeSet(), context);
570  if (unlikely(!ok))
571  {
572  return etiss::RETURNCODE::GENERALERROR;
573  }
574  cb.endaddress_ += mainba.byteCount(); // update end address
575  }
576 
577  count++;
578  } while ((count < maxcount && !context.force_block_end_) || context.force_append_next_instr_);
579 
580  return etiss::RETURNCODE::NOERROR;
581 }

583 void Translation::unloadBlocks(etiss::uint64 startindex, etiss::uint64 endindex)
584 {
585  const etiss::uint64 startindexblock = startindex >> 9;
586  const etiss::uint64 endindexblock = (endindex >> 9) + ((((endindex >> 9) << 9) == endindex) ? 0 : 1);
587  for (etiss::uint64 block = startindexblock; block < endindexblock; block++)
588  {
589  if (blockmap_.empty())
590  break;
591  auto entry = blockmap_.find(block);
592  if (entry != blockmap_.end())
593  {
594  for (std::list<BlockLink *>::iterator iter = entry->second.begin(); iter != entry->second.end();)
595  {
596  BlockLink *bl = *iter;
597  if (bl->start < endindex || bl->end > startindex)
598  {
599  bl->valid = false;
600  BlockLink::updateRef(bl->next, 0);
601  BlockLink::updateRef(bl->branch, 0);
602  entry->second.erase(iter++);
603  BlockLink::decrRef(bl); // remove reference of map
604  }
605  else
606  {
607  iter++;
608  }
609  }
610  if (entry->second.empty())
611  blockmap_.erase(entry);
612  }
613  }
614 }
