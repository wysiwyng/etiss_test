410 void etiss_loadIniConfigs()
411 {
412  if (!po_simpleIni) // no .ini files were given.
413  {
414  return;
415  }
416  std::cout << " Load Configs from .ini files:" << std::endl;
417 
418  // preload loglevel
419  if (!etiss::cfg().isSet("etiss.loglevel"))
420  {
421  etiss::cfg().set<int>("etiss.loglevel", po_simpleIni->GetLongValue("IntConfigurations", "etiss.loglevel", etiss::WARNING));
422  }
423  {
424  int ll = cfg().get<int>("etiss.loglevel", etiss::WARNING);
425  if (ll >= 0 && ll <= etiss::VERBOSE)
426  { // valid log level
427  // dnm
428  // etiss::verbosity() = etiss::VERBOSE;
429  etiss::verbosity() = (Verbosity)ll;
430  etiss::log(etiss::VERBOSE, "Log level set to VERBOSE");
431  }
432  else
433  {
434  etiss::verbosity() = etiss::WARNING;
435  etiss::log(etiss::ERROR, "Specified log level is not valid. must range between 0 (= "
436  "silent) and 5 (= verbose)");
437  }
438  }
439 
440  // get all sections
441  CSimpleIniA::TNamesDepend sections;
442  po_simpleIni->GetAllSections(sections);
443  for (auto iter_section : sections)
444  {
445  // only load config sections
446  if (std::string(iter_section.pItem) != "StringConfigurations" &&
447  std::string(iter_section.pItem) != "BoolConfigurations" &&
448  std::string(iter_section.pItem) != "IntConfigurations")
449  {
450  continue;
451  }
452  etiss::log(etiss::INFO, std::string(" [") + iter_section.pItem + ']');
453 
454  // get all keys in a section
455  CSimpleIniA::TNamesDepend keys;
456  po_simpleIni->GetAllKeys(iter_section.pItem, keys);
457  for (auto iter_key : keys)
458  {
459  std::stringstream message;
460  bool warning = false;
461 
462  // skip loglevel
463  if (std::string(iter_key.pItem) == "etiss.loglevel")
464  continue;
465 
466  // check if cfg is already set
467  if (etiss::cfg().isSet(iter_key.pItem))
468  {
469  message << " cfg already set on command line. ";
470  message << " " << iter_key.pItem << "=";
471  const ::std::type_info& type = vm[std::string(iter_key.pItem)].value().type() ;
472  if (type == typeid(::std::string))
473  message << vm[std::string(iter_key.pItem)].as<std::string>() << ",";
474  else if (type == typeid(int))
475  message << vm[std::string(iter_key.pItem)].as<int>() << ",";
476  else if (type == typeid(bool))
477  message << std::boolalpha << vm[std::string(iter_key.pItem)].as<bool>() << ",";
478  warning = true;
479  }
480  else
481  {
482  // write key (=option) to message.
483  message << " " << iter_key.pItem << "=";
484 
485  // get all values of a key with multiple values
486  CSimpleIniA::TNamesDepend values;
487  po_simpleIni->GetAllValues(iter_section.pItem, iter_key.pItem, values);
488  for (auto iter_value : values)
489  {
490  // Handle configurations
491  if (std::string(iter_section.pItem) == "StringConfigurations")
492  {
493  etiss::cfg().set<std::string>(iter_key.pItem, iter_value.pItem);
494  }
495  else if (std::string(iter_section.pItem) == "BoolConfigurations")
496  {
497  std::string itemval = iter_value.pItem;
498  boost::algorithm::to_lower(itemval); // converts itemval to lower case string
499  bool val;
500 
501  if ((itemval == "true") | (itemval == "yes") | (itemval == "1") | (itemval == "t")) val = true;
502  else if ((itemval == "false") | (itemval == "no") | (itemval == "0") | (itemval == "f")) val = false;
503  else etiss::log(etiss::FATALERROR, std::string("Configuration value ") + iter_key.pItem + " could not be parsed as boolean");
504 
505  etiss::cfg().set<bool>(iter_key.pItem, val);
506  }
507  else if (std::string(iter_section.pItem) == "IntConfigurations") // already load!
508  {
509  std::string itemval = iter_value.pItem;
510  std::size_t sz = 0;
511  long long val;
512  try {
513  val = std::stoll(itemval, &sz, 0);
514  }
515  // catch invalid_argument exception.
516  catch (std::invalid_argument const&){
517  etiss::log(etiss::FATALERROR, std::string("Configuration value ") + iter_key.pItem + " could not be parsed as integer");
518  }
519  etiss::cfg().set<long long>(iter_key.pItem, val);
520  // we use double, as long could have only 32 Bit (e.g. on Windows)
521  // and long long is not offered by the ini library
522  }
523  else
524  // we don't add a DoubleConfigurations section, as converting them
525  // to and from strings could provoke accuracy issues.
526  // To support double, a Configuration::get<double>() has to be
527  // added to Misc.cpp
528  {
529  message << " Section not found for Value:";
530  warning = true;
531  }
532 
533  // write item (=option value) to message.
534  message << iter_value.pItem << ",";
535  }
536 
537  // check if more than one value is set in the ini file
538  if (values.size() > 1)
539  {
540  warning = true;
541  message << " Multi values. Take only LAST one!";
542  }
543  }
544  // add message to etiss log.
545  etiss::log(warning ? etiss::WARNING : etiss::INFO, message.str());
546  }
547  }
