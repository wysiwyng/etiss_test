160 etiss::int32 RISCV64Timer::execute()
161 {
162 
163  etiss::uint64 new_mtime = ((ETISS_CPU *)riscv64cpu)->cpuTime_ps / ((ETISS_CPU *)riscv64cpu)->cpuCycleTime_ps;
164  mtime_overflow_ = (new_mtime < mtime_) ? true : false;
165  mtime_ = new_mtime;
166 
167  if (mtimecmp_buf_[0])
168  {
169 
170  if (!timer_enabled_)
171  timer_enabled_ = true;
172  etiss::uint64 new_mtimecmp = *((etiss::uint64 *)mtimecmp_buf_);
173  bool mtimecmp_overflow_val = (new_mtimecmp < mtimecmp_) ? true : false;
174  if (mtimecmp_overflow_val)
175  {
176  if (mtimecmp_overflow_clear_)
177  {
178  mtimecmp_overflow_clear_ = false;
179  }
180  else
181  mtimecmp_overflow_ = mtimecmp_overflow_val;
182  }
183  mtimecmp_ = new_mtimecmp;
184  memset(mtimecmp_buf_, 0, 8);
185  std::stringstream ss;
186  ss << "Current time: 0x" << std::hex << mtime_ << std::endl;
187  ss << "New timecmp value: 0x" << std::hex << mtimecmp_ << std::endl;
188  etiss::log(etiss::VERBOSE, ss.str());
189  }
190 
191  if (!timer_enabled_)
192  return etiss::RETURNCODE::NOERROR;
193 
194  if (mtimecmp_overflow_ && !mtime_overflow_)
195  {
196  return etiss::RETURNCODE::NOERROR;
197  }
198  else if (mtime_ >= mtimecmp_ || (!mtimecmp_overflow_ && mtime_overflow_))
199  {
200  (riscv64cpu->CSR[CSR_MIP]) |= MIP_MTIP;
201 
202  if (mtimecmp_overflow_)
203  {
204  mtime_overflow_ = false;
205  if (mtimecmp_overflow_)
206  mtimecmp_overflow_ = false;
207  else
208  mtimecmp_overflow_clear_ = true;
209  }
210  return etiss::RETURNCODE::INTERRUPT;
211  }
212  return etiss::RETURNCODE::NOERROR;
213 }
