50 RISCV64MMU::RISCV64MMU(bool pid_enabled) : MMU(true, "RISCV-sv39-MMU", pid_enabled)
51 {
52  // Use builder utility to build up format singleton, bit field for Page Table Entry (PTE)
53  PTEFormatBuilder::Instance()
54  .AddPPNBitField(53, 10)
55  .AddFlag("D", 7)
56  .AddFlag("A", 6)
57  .AddFlag("G", 5)
58  .AddFlag("U", 4)
59  .AddFlag("X", 3)
60  .AddFlag("W", 2)
61  .AddFlag("R", 1)
62  .AddFlag("V", 0)
63  .AddFlag("MODE", 63, 60)
64  .AddFlag("ASID", 59, 44);
65 
66  // Set page size offset
67  PTEFormatBuilder::Instance().AddPageOffset(11, 0);
68 
69  // Convenient manipulation for Virtual Address (VM), not part of PTE
70  PTEFormatBuilder::Instance().AddFlag("VPN2", 38, 30).AddFlag("VPN1", 29, 21).AddFlag("VPN0", 20, 12);
71 
72  // Convenient manipulation for Physical Address (PM), not part of PTE
73  PTEFormatBuilder::Instance().AddFlag("PPN2", 53, 28).AddFlag("PPN1", 27, 19).AddFlag("PPN0", 18, 10);
74 
75  // Allow mapping several vma to same pma
76  REGISTER_PAGE_FAULT_HANDLER(PTEOVERLAP, tlb_overlap_handler);
77 }

79 int32_t RISCV64MMU::WalkPageTable(uint64_t vma, MM_ACCESS access)
80 {
81 
82  if (mmu_enabled_)
83  {
84  // Initialize leaf PTE
85  PTE leaf_pte = PTE(0);
86  PTE vma_pte = PTE(vma);
87  // Get the PPN of page global directory
88  uint64_t addr_ppn = GetMMUControlReg() & SATP_PPN_MASK;
89  // Walk through page directory to find Page Table Entry (PTE)
90  uint64_t leaf_pte_val = 0;
91  unsigned char *buffer = (unsigned char *)(&leaf_pte_val);
92  int32_t i = LEVELS;
93  uint64_t addr = 0;
94  uint64_t addr_offset = 0;
95  int32_t fault = etiss::RETURNCODE::NOERROR;
96 
97  while (1)
98  {
99  // Point to current vpn index
100  --i;
101  std::stringstream vpn_name;
102  vpn_name << "VPN" << i;
103  addr_offset = vma_pte.GetByName(vpn_name.str());
104  addr = (addr_ppn << PAGE_OFFSET) + addr_offset * PTESIZE;
105 
106  if ((fault = system_->dread(system_->handle, cpu_, addr, buffer, PTESIZE)))
107  return fault;
108  // A new leaf pte value is read from page directory
109  leaf_pte.Update(leaf_pte_val);
110 
111  if ((0 == leaf_pte.GetByName("V")) || ((0 == leaf_pte.GetByName("R")) && (1 == leaf_pte.GetByName("W"))))
112  {
113  etiss::log(etiss::INFO, GetName() + ": PTE invalid");
114  goto RETURN_PAGEFAULT;
115  }
116  if ((1 == leaf_pte.GetByName("R")) || (1 == leaf_pte.GetByName("X")))
117  break;
118  if (0 > i)
119  {
120  etiss::log(etiss::INFO, GetName() + ": Found no valid pte in page table");
121  goto RETURN_PAGEFAULT;
122  }
123  // The leaf node turn out to be a page table/page directory, go to the next level with new ppn.
124  addr_ppn = leaf_pte.GetByName("PPN");
125  }
126 
127  // TODO: MXR, SUM check should be implemented later on.
128 
129  for (int tmp_i = i; tmp_i > 0; --tmp_i)
130  {
131  }
132 
133  if (0 == leaf_pte.GetByName("A"))
134  {
135  leaf_pte_val |= PTE_A;
136  leaf_pte.Update(leaf_pte_val);
137  if ((fault = system_->dwrite(system_->handle, cpu_, addr, buffer, PTESIZE)))
138  return fault;
139  }
140 
141  if ((0 == leaf_pte.GetByName("A")) && (W_ACCESS == access))
142  {
143  leaf_pte_val |= PTE_D;
144  leaf_pte.Update(leaf_pte_val);
145  if ((fault = system_->dwrite(system_->handle, cpu_, addr, buffer, PTESIZE)))
146  return fault;
147  }
148 
149  // TODO: PMA, PMP check should be implemented later on.
150 
151  uint64_t new_pte_val = leaf_pte.Get();
152 
153  // Superpage translation
154  if (i > 0)
155  {
156  for (int tmp_i = i; tmp_i-- > 0; --tmp_i)
157  {
158  // Check misagligned superpage
159  std::stringstream ppn_name;
160  ppn_name << "PPN" << tmp_i;
161  if (0 != leaf_pte.GetByName(ppn_name.str()))
162  {
163  etiss::log(etiss::WARNING, GetName() + ": Superpage misagligned");
164  goto RETURN_PAGEFAULT;
165  }
166  std::stringstream vpn_name;
167  vpn_name << "VPN" << tmp_i;
168  // Debug +10 is just a work around
169  new_pte_val |= vma_pte.GetByName(vpn_name.str()) << ((VPN_OFFSET * tmp_i) + 10);
170  }
171  }
172  PTE new_pte = PTE(new_pte_val);
173  AddTLBEntry(vma >> PAGE_OFFSET, new_pte);
174  // Map TLB entry to physical address for write-through, because TLB is more or less a cache
175  AddTLBEntryMap(addr, new_pte);
176  }
177 
178  return etiss::RETURNCODE::NOERROR;
179 
180 RETURN_PAGEFAULT:
181  switch (access)
182  {
183  case R_ACCESS:
184  return etiss::RETURNCODE::LOAD_PAGEFAULT;
185  case W_ACCESS:
186  return etiss::RETURNCODE::STORE_PAGEFAULT;
187  case X_ACCESS:
188  return etiss::RETURNCODE::INSTR_PAGEFAULT;
189  default:
190  abort();
191  }
192 }

194 int32_t RISCV64MMU::CheckProtection(const PTE &pte, MM_ACCESS access)
195 {
196  switch (access)
197  {
198  case R_ACCESS:
199  if (1 == pte.GetByName("R"))
200  break;
201  return etiss::RETURNCODE::LOAD_PAGEFAULT;
202  case W_ACCESS:
203  if (1 == pte.GetByName("W"))
204  break;
205  return etiss::RETURNCODE::STORE_PAGEFAULT;
206  case X_ACCESS:
207  if (1 == pte.GetByName("X"))
208  break;
209  return etiss::RETURNCODE::INSTR_PAGEFAULT;
210  }
211  return etiss::RETURNCODE::NOERROR;
212 }
