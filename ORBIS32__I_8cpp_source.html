48 static void l_addsubX(std::string in1, std::string in2, std::string out, CodeSet &codeset, bool add, bool cy)
49 {
50  // add/sub
51  {
52  CodePart &part = codeset.append(CodePart::INITIALREQUIRED);
53  part.getCode() = "\t\tetiss_uint64 a = (uint32_t) " + in1 + ";\n\t\tetiss_uint64 b = (uint32_t) " + in2 +
54  ";\n\t\tetiss_uint64 d = a " + (add ? "+" : "-") + " b;" + out + " = (etiss_uint32)d;\n";
55  if (cy)
56  {
57  part.getRegisterDependencies().add("SR", 32, OR1K_SR_CY);
58  part.getCode() = part.getCode() + "\t\tif (((*((OR1K*)cpu)->SR)&OR1K_SR_CY) != 0) {" + out + " = " + out +
59  " " + (add ? "+" : "-") + " 1; d = d " + (add ? "+" : "-") + "1 ;}";
60  }
61  }
62 
63  // carry
64  {
65  CodePart &part = codeset.append(CodePart::OPTIONALMIDDLE);
66  part.getAffectedRegisters().add("SR", 32, OR1K_SR_CY);
67  part.getCode() = "\t\tif (unlikely(d != (uint64_t)((uint32_t)(" + out +
68  ")))) (*((OR1K*)cpu)->SR) |= OR1K_SR_CY; else (*((OR1K*)cpu)->SR) &= ~OR1K_SR_CY;";
69  }
70 
71  // overflow
72  {
73  CodePart &part = codeset.append(CodePart::OPTIONALMIDDLE);
74  part.getAffectedRegisters().add("SR", 32, OR1K_SR_OV);
75  part.getCode() = std::string("\t\tif (unlikely(((a&(1<<31)) ") + (add ? "==" : "!=") +
76  " (b&(1<<31))) && ((a&(1<<31)) != (d&(1<<31))))) (*((OR1K*)cpu)->SR) |= OR1K_SR_OV; else "
77  "(*((OR1K*)cpu)->SR) &= ~OR1K_SR_OV;";
78  }
79 
80  // exception
81 
82  CodePart &part = codeset.append(CodePart::APPENDEDRETURNINGREQUIRED);
83  part.getAffectedRegisters().add("SR", 32, OR1K_SR_OV);
84  part.getAffectedRegisters().add("SR", 32, OR1K_SR_CY);
85  part.getRegisterDependencies().add("NPC", 32);
86  part.getRegisterDependencies().add("PPC", 32);
87  part.getRegisterDependencies().add("instructionPointer", 32);
88  std::stringstream ss;
89 
90  ss << "\t\tif (unlikely((*((OR1K*)cpu)->SR)&OR1K_SR_OVE)){\n"
91  "\t\t\tif (unlikely((*((OR1K*)cpu)->AECR) & (OR1K_AECR_CYADDE | OR1K_AECR_OVADDE))){\n"
92  "\t\t\t\tif (((a&(1<<31)) == (b&(1<<31))) && ((a&(1<<31)) != (d&(1<<31)))) (*((OR1K*)cpu)->SR) |= "
93  "OR1K_SR_OV; else (*((OR1K*)cpu)->SR) &= ~OR1K_SR_OV;\n"
94  "\t\t\t\tif (d != (" +
95  out +
96  ")) (*((OR1K*)cpu)->SR) |= OR1K_SR_CY; else (*((OR1K*)cpu)->SR) &= ~OR1K_SR_CY;\n"
97  "\t\t\t\tif ((*((OR1K*)cpu)->SR) & (OR1K_SR_CY | OR1K_SR_OV)) return OR1K_RANGEEXCEPTION;\n"
98  "\t\t\t}\n"
99  "\t\t}";
100  part.getCode() = ss.str();
101 }

103 template <char op>
104 static bool simple_op_rDAB(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
105 {
106  unsigned irD = rd.read(ba);
107  unsigned irA = ra.read(ba);
108  unsigned irB = rb.read(ba);
109  CodePart &part = cs.append(CodePart::INITIALREQUIRED);
110  part.getRegisterDependencies().add(reg_name[irA], 32);
111  part.getRegisterDependencies().add(reg_name[irB], 32);
112  part.getAffectedRegisters().add(reg_name[irD], 32);
113  part.code() = std::string(reg[irD]) + " = " + reg[irA] + op + reg[irB] + ";";
114 
115  update_pc<false, false>(ba, cs, ic);
116 
117  return true;
118 }
119 template <char op, bool sig>
120 static bool simple_op_rDAI(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
121 {
122  typedef typename std::conditional<sig, int16_t, uint16_t>::type IMMT;
123  unsigned irD = rd.read(ba);
124  unsigned irA = ra.read(ba);
125  IMMT irI = immediate.read(ba);
126  CodePart &part = cs.append(CodePart::INITIALREQUIRED);
127  part.getRegisterDependencies().add(reg_name[irA], 32);
128  part.getAffectedRegisters().add(reg_name[irD], 32);
129  part.code() = std::string(reg[irD]) + " = " + reg[irA] + op + " " + toString(irI) + ";";
130 
131  update_pc<false, false>(ba, cs, ic);
132 
133  return true;
134 }

136 template <bool right, bool sig, bool imm>
137 static bool shift(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
138 {
139  unsigned irD = rd.read(ba);
140  unsigned irA = ra.read(ba);
141  unsigned irB = rb.read(ba);
142  uint16_t imme = (uint16_t)(immediate.read(ba) & 0x1f);
143  CodePart &part = cs.append(CodePart::INITIALREQUIRED);
144  part.getRegisterDependencies().add(reg_name[irA], 32);
145  part.getRegisterDependencies().add(reg_name[irB], 32);
146  part.getAffectedRegisters().add(reg_name[irD], 32);
147  part.code() = std::string(reg[irD]) + " = ((" + (sig ? "etiss_int32)" : "etiss_uint32)") + reg[irA] +
148  (right ? ")>>(" : ")<<(") + (imm ? toString(imme) : (std::string(reg[irB]) + "&0x1f")) + ");";
149 
150  update_pc<false, false>(ba, cs, ic);
151 
152  return true;
153 }

155 template <typename T, bool swap = (ETISS_ENDIANNESS != ETISS_BIG_ENDIAN)>
156 static bool loadinstr(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
157 {
158 
159  std::stringstream ss;
160  ss << "etiss_uint32 * or1k_rD = &" << reg[rd.read(ba)] << ";";
161  ss << "etiss_uint8 * or1k_buf = (etiss_uint8 *)or1k_rD;\n"
162  "etiss_int32 or1k_exception = (*(system->dread))(system->handle,cpu,("
163  << reg[ra.read(ba)] << "+" << (int16_t)immediate.read(ba) << "),or1k_buf," << sizeof(T) << ");\n";
164  if (swap)
165  {
166  if (sizeof(T) >= 2)
167  ss << "etiss_uint8 or1k_tmpswap;\n";
168  for (unsigned i = 0; i < (sizeof(T) >> 1); i++)
169  {
170  ss << "or1k_tmpswap = or1k_buf[" << i << "]; or1k_buf[" << i << "] = or1k_buf[" << (sizeof(T) - 1 - i)
171  << "]; or1k_buf[" << (sizeof(T) - 1 - i) << "] = or1k_tmpswap;";
172  }
173  }
174  if (std::is_signed<T>::value)
175  {
176  if (sizeof(T) == 1)
177  {
178  ss << "*or1k_rD = (int8_t) (*or1k_rD);";
179  }
180  if (sizeof(T) == 2)
181  {
182  ss << "*or1k_rD = (int16_t)(*or1k_rD);";
183  }
184  }
185  else
186  {
187  if (sizeof(T) == 1)
188  {
189  ss << "*or1k_rD = (uint8_t) (*or1k_rD);";
190  }
191  if (sizeof(T) == 2)
192  {
193  ss << "*or1k_rD = (uint16_t)(*or1k_rD);";
194  }
195  }
196  CodePart &cp = cs.append(CodePart::INITIALREQUIRED);
197  cp.code() = ss.str();
198  cp.getAffectedRegisters().add(reg_name[rd.read(ba)], 32);
199  cp.getRegisterDependencies().add(reg_name[ra.read(ba)], 32);
200  cs.append(CodePart::APPENDEDRETURNINGREQUIRED).code() = "if (or1k_exception != 0) \n\treturn or1k_exception;";
201  update_pc<false, false>(ba, cs, ic);
202  return true;
203 }

205 template <typename T, bool swap = (ETISS_ENDIANNESS != ETISS_BIG_ENDIAN)>
206 static bool storeinstr(BitArray &ba, etiss::CodeSet &cs, InstructionContext &ic)
207 {
208 
209  BitArrayRange immp1(25, 21);
210  BitArrayRange immp2(10, 0);
211  std::stringstream ss;
212  ss << "etiss_uint32 or1k_rB = " << reg[rb.read(ba)] << ";";
213  ss << "etiss_uint8 * or1k_buf = (etiss_uint8 *)&or1k_rB;\n";
214  if (swap)
215  {
216  if (sizeof(T) >= 2)
217  ss << "etiss_uint8 or1k_tmpswap;";
218  for (unsigned i = 0; i < (sizeof(T) >> 1); i++)
219  {
220  ss << "or1k_tmpswap = or1k_buf[" << i << "]; or1k_buf[" << i << "] = or1k_buf[" << (sizeof(T) - 1 - i)
221  << "]; or1k_buf[" << (sizeof(T) - 1 - i) << "] = or1k_tmpswap;";
222  }
223  }
224  ss << "etiss_int32 or1k_exception = (*(system->dwrite))(system->handle,cpu,(" << reg[ra.read(ba)] << "+"
225  << toString((int16_t)((immp1.read(ba) << 11) | immp2.read(ba))) << "),or1k_buf," << sizeof(T) << ");\n";
226  CodePart &cp = cs.append(CodePart::INITIALREQUIRED);
227  cp.code() = ss.str();
228  cp.getAffectedRegisters().add(reg_name[rd.read(ba)], 32);
229  cp.getRegisterDependencies().add(reg_name[ra.read(ba)], 32);
230  cs.append(CodePart::APPENDEDRETURNINGREQUIRED).code() = "if (or1k_exception != 0) \n\treturn or1k_exception;";
231  update_pc<false, false>(ba, cs, ic);
232  return true;
233 }
