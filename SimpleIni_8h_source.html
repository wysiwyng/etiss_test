1625 template <class SI_CHAR, class SI_STRLESS, class SI_CONVERTER>
1626 bool CSimpleIniTempl<SI_CHAR, SI_STRLESS, SI_CONVERTER>::LoadMultiLineText(SI_CHAR *&a_pData, const SI_CHAR *&a_pVal,
1627  const SI_CHAR *a_pTagName,
1628  bool a_bAllowBlankLinesInComment) const
1629 {
1630  // we modify this data to strip all newlines down to a single '\n'
1631  // character. This means that on Windows we need to strip out some
1632  // characters which will make the data shorter.
1633  // i.e. LINE1-LINE1\r\nLINE2-LINE2\0 will become
1634  // LINE1-LINE1\nLINE2-LINE2\0
1635  // The pDataLine entry is the pointer to the location in memory that
1636  // the current line needs to start to run following the existing one.
1637  // This may be the same as pCurrLine in which case no move is needed.
1638  SI_CHAR *pDataLine = a_pData;
1639  SI_CHAR *pCurrLine;
1640 
1641  // value starts at the current line
1642  a_pVal = a_pData;
1643 
1644  // find the end tag. This tag must start in column 1 and be
1645  // followed by a newline. No whitespace removal is done while
1646  // searching for this tag.
1647  SI_CHAR cEndOfLineChar = *a_pData;
1648  for (;;)
1649  {
1650  // if we are loading comments then we need a comment character as
1651  // the first character on every line
1652  if (!a_pTagName && !IsComment(*a_pData))
1653  {
1654  // if we aren't allowing blank lines then we're done
1655  if (!a_bAllowBlankLinesInComment)
1656  {
1657  break;
1658  }
1659 
1660  // if we are allowing blank lines then we only include them
1661  // in this comment if another comment follows, so read ahead
1662  // to find out.
1663  SI_CHAR *pCurr = a_pData;
1664  int nNewLines = 0;
1665  while (IsSpace(*pCurr))
1666  {
1667  if (IsNewLineChar(*pCurr))
1668  {
1669  ++nNewLines;
1670  SkipNewLine(pCurr);
1671  }
1672  else
1673  {
1674  ++pCurr;
1675  }
1676  }
1677 
1678  // we have a comment, add the blank lines to the output
1679  // and continue processing from here
1680  if (IsComment(*pCurr))
1681  {
1682  for (; nNewLines > 0; --nNewLines)
1683  *pDataLine++ = '\n';
1684  a_pData = pCurr;
1685  continue;
1686  }
1687 
1688  // the comment ends here
1689  break;
1690  }
1691 
1692  // find the end of this line
1693  pCurrLine = a_pData;
1694  while (*a_pData && !IsNewLineChar(*a_pData))
1695  ++a_pData;
1696 
1697  // move this line down to the location that it should be if necessary
1698  if (pDataLine < pCurrLine)
1699  {
1700  size_t nLen = (size_t)(a_pData - pCurrLine);
1701  memmove(pDataLine, pCurrLine, nLen * sizeof(SI_CHAR));
1702  pDataLine[nLen] = '\0';
1703  }
1704 
1705  // end the line with a NULL
1706  cEndOfLineChar = *a_pData;
1707  *a_pData = 0;
1708 
1709  // if are looking for a tag then do the check now. This is done before
1710  // checking for end of the data, so that if we have the tag at the end
1711  // of the data then the tag is removed correctly.
1712  if (a_pTagName && (!IsLess(pDataLine, a_pTagName) && !IsLess(a_pTagName, pDataLine)))
1713  {
1714  break;
1715  }
1716 
1717  // if we are at the end of the data then we just automatically end
1718  // this entry and return the current data.
1719  if (!cEndOfLineChar)
1720  {
1721  return true;
1722  }
1723 
1724  // otherwise we need to process this newline to ensure that it consists
1725  // of just a single \n character.
1726  pDataLine += (a_pData - pCurrLine);
1727  *a_pData = cEndOfLineChar;
1728  SkipNewLine(a_pData);
1729  *pDataLine++ = '\n';
1730  }
1731 
1732  // if we didn't find a comment at all then return false
1733  if (a_pVal == a_pData)
1734  {
1735  a_pVal = NULL;
1736  return false;
1737  }
1738 
1739  // the data (which ends at the end of the last line) needs to be
1740  // null-terminated BEFORE before the newline character(s). If the
1741  // user wants a new line in the multi-line data then they need to
1742  // add an empty line before the tag.
1743  *--pDataLine = '\0';
1744 
1745  // if looking for a tag and if we aren't at the end of the data,
1746  // then move a_pData to the start of the next line.
1747  if (a_pTagName && cEndOfLineChar)
1748  {
1749  SI_ASSERT(IsNewLineChar(cEndOfLineChar));
1750  *a_pData = cEndOfLineChar;
1751  SkipNewLine(a_pData);
1752  }
1753 
1754  return true;
1755 }
