100  {
101  for (size_t i = 0; i < (1 << log2_buckets); i++)
102  {

105  // memset(map,0,sizeof(Entry*)*(1<<log2_buckets));

109  for (size_t i = 0; i < (1 << log2_buckets); i++)
110  {
111  if (map[i])
112  {
113  Entry *del = map[i];
114  do
115  {
116  pool.destroy(del);
117  pool.deallocate(del);

119  del = n;
120  } while (del);
121  }
122  }
