155 void SimpleMemSystem::load_elf()
156 {
157  if (!etiss::cfg().isSet("vp.elf_file")) return;
158 
159  std::string elf_file = etiss::cfg().get<std::string>("vp.elf_file", "");
160 
161  ELFIO::elfio reader;
162 
163  if (!reader.load(elf_file))
164  {
165  etiss::log(etiss::FATALERROR, "ELF reader could not process file");
166  }
167 
168  if (etiss::cfg().isSet("arch.cpu")) {
169  std::stringstream ss;
170  ss << "Assuming CPU architecture " << etiss::cfg().get<std::string>("arch.cpu", "") << " as set in configuration file. ELF architecture field will be ignored";
171  etiss::log(etiss::INFO, ss.str());
172  } else {
173  // set architecture automatically
174  if (reader.get_machine() == EM_RISCV)
175  {
176  if ((reader.get_class() == ELFCLASS64)) {
177  etiss::cfg().set<std::string>("arch.cpu", "RISCV64"); // RISCV and OR1K work as well
178  } else if ((reader.get_class() == ELFCLASS32)) {
179  etiss::cfg().set<std::string>("arch.cpu", "RV32IMACFD");
180  // add conditions
181  } else {
182  etiss::log(etiss::FATALERROR, "System architecture is neither 64 nor 32 bit!");
183  }
184  }
185  else if (reader.get_machine() == EM_OPENRISC)
186  {
187  if ((reader.get_class() == ELFCLASS32))
188  etiss::cfg().set<std::string>("arch.cpu", "OR1K");
189  if ((reader.get_class() == ELFCLASS64))
190  etiss::log(etiss::FATALERROR, "OR1k 64 is not supported");
191  }
192  else
193  {
194  std::stringstream ss;
195  ss << "Target architecture with code 0x" << std::hex << std::setw(2) << std::setfill('0') << reader.get_machine() << " was not automatically recognized, please set the arch.cpu parameter manually!";
196  etiss::log(etiss::FATALERROR, ss.str());
197  }
198  std::stringstream ss;
199  ss << "Set ETISS architecture to " << etiss::cfg().get<std::string>("arch.cpu", "") << " as specified in ELF-file.";
200  etiss::log(etiss::INFO, ss.str());
201  }
202 
203  for (auto &seg : reader.segments)
204  {
205  etiss::uint64 start_addr = seg->get_physical_address();
206  etiss::uint64 size = seg->get_memory_size();
207  if (size == 0) continue;
208  size_t file_size = seg->get_file_size();
209  if (seg->get_type() != PT_LOAD) continue;
210 
211  int mode = 0;
212  std::string modestr = "";
213  if (seg->get_flags() & PF_R) {
214  mode |= MemSegment::READ;
215  modestr += "R";
216  }
217  if (seg->get_flags() & PF_W) {
218  mode |= MemSegment::WRITE;
219  modestr += "W";
220  }
221  if (seg->get_flags() & PF_X) {
222  mode |= MemSegment::EXEC;
223  modestr += "X";
224  }
225 
226  std::stringstream sname;
227  sname << seg->get_index() << " - " << modestr
228  << "[0x" << std::hex << std::setfill('0') << std::setw(sizeof(etiss::uint64) * 2) << start_addr << " - "
229  << "0x" << std::hex << std::setfill('0') << std::setw(sizeof(etiss::uint64) * 2) << start_addr + size - 1 << "]";
230 
231  auto mseg_it = std::find_if(msegs_.begin(), msegs_.end(), find_fitting_mseg(start_addr, size));
232 
233  if (mseg_it != msegs_.end()) {
234  auto & mseg = *mseg_it;
235 
236  mseg->name_ = sname.str();
237  mseg->mode_ = static_cast<MemSegment::access_t>(mode);
238 
239  mseg->load(seg->get_data(), start_addr - mseg->start_addr_, file_size);
240 
241  std::stringstream msg;
242  msg << "Initialized the memory segment " << mseg->name_ << " from ELF-file";
243  etiss::log(etiss::INFO, msg.str());
244 
245  continue;
246  }
247 
248  std::stringstream msg;
249  msg << "Found no matching memory segments at 0x" << std::hex << std::setfill('0') << std::setw(8) << start_addr;
250 
251  if (error_on_seg_mismatch_) {
252  msg << "! As you turned on error_on_seg_mismatch, ETISS will now terminate.";
253  etiss::log(etiss::FATALERROR, msg.str());
254  } else {
255  msg << ", creating one. WARNING: the segment will be created with the size information present in the ELF-file, the resulting segment may be too small to fit dynamic data (cache, heap)!";
256  etiss::log(etiss::WARNING, msg.str());
257  }
258 
259  auto mseg = std::make_unique<MemSegment>(start_addr, size, static_cast<MemSegment::access_t>(mode), sname.str());
260  add_memsegment(mseg, seg->get_data(), file_size);
261  }
262 
263  // read start or rather program boot address from ELF
264  start_addr_ = reader.get_entry();
265 }

336 etiss::int32 SimpleMemSystem::dbus_access(ETISS_CPU *cpu, etiss::uint64 addr, etiss::uint8 *buf, etiss::uint32 len) {
337  auto mseg_it = std::find_if(msegs_.begin(), msegs_.end(), find_fitting_mseg(addr, len));
338 
339  if (mseg_it != msegs_.end()) {
340  auto & mseg = *mseg_it;
341  MemSegment::access_t access = write ? MemSegment::WRITE : MemSegment::READ;
342 
343  if (!(mseg->mode_ & access)) {
344  access_error(cpu, addr, len, std::string("dbus ") + (write ? "write" : "read") + " forbidden", etiss::WARNING);
345  }
346 
347  size_t offset = addr - mseg->start_addr_;
348 
349  void * dest = write ? mseg->mem_ + offset : buf;
350  const void * src = write ? buf : mseg->mem_ + offset;
351 
352  memcpy(dest, src, len);
353 
354  if (print_dbus_access_) trace(cpu, addr, len, write, print_to_file_, trace_file_dbus_);
355 
356  return RETURNCODE::NOERROR;
357  }
358 
359  access_error(cpu, addr, len, std::string("dbus ") + (write ? "write" : "read") + " error", etiss::ERROR);
360 
361  return write ? RETURNCODE::DBUS_WRITE_ERROR : RETURNCODE::DBUS_READ_ERROR;
362 }
