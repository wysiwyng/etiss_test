572 static void HandleCollision(Instruction *instr, Node *&node)
573 {
574  auto otherInstr = dynamic_cast<Instruction*>(node);
575  if (otherInstr)
576  {
577  AssertOnTrueCollision(instr, otherInstr);
578  if ((instr->opc_.mask_ & otherInstr->opc_.mask_) == instr->opc_.mask_)
579  {
580  node = new OverlappedNode(otherInstr, instr);
581  } else {
582  node = new OverlappedNode(instr, node);
583  }
584  } else {
585  auto otherNode = dynamic_cast<OverlappedNode*>(node);
586  if (otherNode)
587  {
588  AssertOnTrueCollision(instr, otherNode->m_matchInstr);
589  if ((instr->opc_.mask_ & otherNode->m_matchInstr->opc_.mask_) == instr->opc_.mask_)
590  {
591  auto replacedNode = otherNode->m_delegateNode;
592  auto replInstr = dynamic_cast<Instruction*>(replacedNode);
593  BitArray replacedMask(instr->opc_.mask_.width());
594  if (replInstr)
595  {
596  AssertOnTrueCollision(instr, replInstr);
597  replacedMask = replInstr->opc_.mask_;
598  } else {
599  auto replSpecNode = dynamic_cast<OverlappedNode*>(replacedNode);
600  if (replSpecNode)
601  {
602  replacedMask = replSpecNode->m_matchInstr->opc_.mask_;
603  } else {
604  etiss::log(etiss::FATALERROR, "OverlappedNode delegate must not be a plain Node");
605  }
606  }
607  if ((instr->opc_.mask_ & replacedMask) == instr->opc_.mask_)
608  {
609  if (replInstr) {
610  otherNode->m_delegateNode = new OverlappedNode(replInstr, instr);
611  } else {
612  auto replSpecNode = dynamic_cast<OverlappedNode *>(replacedNode);
613  if (replSpecNode)
614  {
615  otherNode->m_delegateNode = new OverlappedNode(instr, replSpecNode);
616  } else {
617  etiss::log(etiss::FATALERROR, "OverlappedNode delegate must not be a plain Node");
618  }
619  }
620  } else {
621  otherNode->m_delegateNode = new OverlappedNode(instr, replacedNode);
622  }
623  } else {
624  node = new OverlappedNode(instr, otherNode);
625  }
626  } else {

631 void Node::compile(const std::map<const OPCode *, Instruction *, etiss::instr::less> &instrmap_, const BitArray &used,
632  bool &ok, std::list<std::string> &warnings, std::list<std::string> &errors)
633 {
634  using MapType = std::remove_const<std::remove_reference<decltype(instrmap_)>::type>::type;
635 
636  // This masks bits that have already been compiled.
637  BitArray mask(~used);
638  // Build mask that has only the bits set that are present in all masks.
639  for (const auto &op2instr : instrmap_)
640  {
641  mask = mask & op2instr.first->mask_;
642  }
643 
644  for (unsigned i = 0; i < mask.width(); i++)
645  {
646  unsigned bsc = mask.getBitSetCount(i);
647  bsc = std::min(bsc, (unsigned)(sizeof(I) * 8) - 1);
648  if (bsc == 0)
649  continue;
650 
651  reader_ = new BitArrayRange(i + bsc - 1, i);
652  size_t size = static_cast<size_t>(1) << bsc;
653  subs_ = new Node *[size]();
654  auto sub_maps = new MapType *[size]();
655 
656  // Only set the bits from the current bit set being examined.
657  mask.setAll(false);
658  reader_->setAll(mask, true);
659  const BitArray newused(used | mask);
660 
661  for (const auto &op2instr : instrmap_)
662  {
663  auto &op = op2instr.first;
664  auto &instr = op2instr.second;
665  auto &node = subs_[reader_->read(op->code_)];
666  auto &sub_map = sub_maps[reader_->read(op->code_)];
667 
668  // Is this a finalized instruction?
669  if (op->mask_ == newused)
670  {
671  // Check if there was already a node.
672  if (node)
673  {
674  // Try to convert existing node to overlap node.
675  if (!sub_map || sub_map->empty())
676  {
677  etiss::log(etiss::FATALERROR, "Expecting non-empty sub_map for existing node");
678  }
679  else
680  {
681  Node *replaceNode = sub_map->begin()->second;
682  for (const auto &subInstr : *sub_map)
683  {
684  HandleCollision(subInstr.second, replaceNode);
685  }
686  HandleCollision(instr, replaceNode);
687  delete node;
688  node = replaceNode;
689  }
690  }
691  else
692  {
693  node = instr;
694  }
695  }
696  else
697  {
698  if (!sub_map)
699  {
700  if (node)
701  {
702  HandleCollision(instr, node);
703  }
704  else
705  {
706  node = new Node();
707  sub_map = new MapType();
708  }
709  }
710  if (sub_map)
711  {
712  if (!sub_map->insert(op2instr).second)
713  {
714  etiss_log(ERROR, "failed to store instruction in map");
715  }
716  }
717  }
718  }
719 
720  for (size_t j = 0; j < size; j++)
721  {
722  if (sub_maps[j])
723  {
724  subs_[j]->compile(*sub_maps[j], newused, ok, warnings, errors);
725  delete sub_maps[j];
726  }
727  }
728  delete[] sub_maps;
729  // Done with the given tree.
730  return;
731  }
732 
733  ok = true;
734 }

827 {

836  {

852  {

854  }
